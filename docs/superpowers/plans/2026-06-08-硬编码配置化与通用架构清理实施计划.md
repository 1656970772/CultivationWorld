# 硬编码配置化与通用架构清理 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 消除首版运行主链中的高风险硬编码、旧回退路径和非通用实现，让运行时数据加载、GAS、行为树、势力 AI、宏观资源和编辑器数据源都回到可配置、可校验、可扩展的正式架构。

**Architecture:** 采用“清单驱动加载 + 注册表 + 策略/规则链 + strict 校验”的整理路线。`DataManifest` 保证目录级配置会被加载，`validateGameData()` 在启动和工具测试中阻止静默缺配置，GAS 统一由 GE/GA/EffectEngine 结算，AI 与编辑器不再靠固定 ID、固定列表或旧数据镜像扩展。

**Tech Stack:** JavaScript ES modules、JSON 数据配置、Node.js 工具测试、Tauri/Rust 编辑器后端、现有 WorldEngine / GAS / Job-Toil / EconomicSystem / RelationshipPlatform。

**最后更新:** 2026-06-08

```javascript
#!/usr/bin/env node
import { readdirSync, readFileSync, existsSync } from 'node:fs';
import { resolve } from 'node:path';

const GAME_ROOT = resolve(new URL('..', import.meta.url).pathname);
const manifestPath = resolve(GAME_ROOT, 'data/config/data-manifest.json');
if (!existsSync(manifestPath)) {
  console.error('FAIL missing data/config/data-manifest.json');
  process.exit(1);
}

const manifest = JSON.parse(readFileSync(manifestPath, 'utf8'));
let failed = 0;
function fail(msg) {
  console.error(`FAIL ${msg}`);
  failed++;
}

function jsonFiles(dir) {
  return readdirSync(resolve(GAME_ROOT, dir)).filter(name => name.endsWith('.json')).sort();
}

for (const group of manifest.groups || []) {
  if (!group.dir || !Array.isArray(group.files) || !group.outputKey) fail(`bad manifest group ${JSON.stringify(group)}`);
  const disk = jsonFiles(group.dir);
  const listed = [...group.files].sort();
  for (const name of disk) {
    if (!listed.includes(name)) fail(`${group.dir}/${name} exists on disk but is not listed in manifest`);
  }
  for (const name of listed) {
    if (!disk.includes(name)) fail(`${group.dir}/${name} is listed in manifest but missing on disk`);
  }
}

const requiredGroups = ['itemDefs', 'effects', 'abilities', 'jobs', 'toils', 'behaviorTrees'];
const outputKeys = new Set((manifest.groups || []).map(g => g.outputKey));
for (const key of requiredGroups) {
  if (!outputKeys.has(key)) fail(`manifest missing group outputKey=${key}`);
}

if (failed > 0) {
  console.error(`\nDataManifest 覆盖测试失败：${failed} 项`);
  process.exit(1);
}
console.log('DataManifest 覆盖测试通过');
```

```javascript
async function loadJSON(path) {
  const resp = await fetch(path);
  if (!resp.ok) throw new Error(`加载配置失败 [${path}]: HTTP ${resp.status}`);
  return resp.json();
}

function camelSuffix(name) {
  return name.replace(/-([a-z])/g, (_, ch) => ch.toUpperCase());
}

export async function loadGameDataManifest(path = 'data/config/data-manifest.json') {
  return loadJSON(path);
}

export async function loadSingletons(manifest) {
  const entries = await Promise.all(
    (manifest.singletons || []).map(async item => [item.key, await loadJSON(item.path)]),
  );
  const out = Object.fromEntries(entries);
  for (const name of manifest.balance || []) {
    const data = await loadJSON(`data/balance/${name}.json`);
    const key = name === 'monster-spawn'
      ? 'monsterSpawn'
      : `balance${camelSuffix(name).replace(/^./, c => c.toUpperCase())}`;
    out[key] = data;
  }
  return out;
}

export async function loadJsonGroup(group) {
  const loaded = await Promise.all(
    group.files.map(async fileName => ({
      fileName,
      data: await loadJSON(`${group.dir}/${fileName}`),
    })),
  );
  if (!group.arrayKey) {
    return Object.fromEntries(loaded.map(item => [item.fileName.replace(/\.json$/i, ''), item.data]));
  }
  return {
    [group.arrayKey]: loaded.flatMap(item => item.data?.[group.arrayKey] || []),
  };
}

export async function loadGroups(manifest) {
  const entries = await Promise.all(
    (manifest.groups || []).map(async group => [group.outputKey, await loadJsonGroup(group)]),
  );
  return Object.fromEntries(entries);
}

export async function loadRelationshipPlatform(manifest) {
  const rel = manifest.relationshipPlatform || {};
  const schemas = {};
  for (const [key, path] of Object.entries(rel.schemas || {})) schemas[key] = await loadJSON(path);
  const dictionaries = {};
  for (const [key, path] of Object.entries(rel.dictionaries || {})) dictionaries[key] = await loadJSON(path);
  const eventHooks = await Promise.all((rel.eventHooks || []).map(loadJSON));
  const impactRules = await Promise.all((rel.impactRules || []).map(loadJSON));
  const signalRules = await Promise.all((rel.signalRules || []).map(loadJSON));
  const groups = rel.groups ? await loadJSON(rel.groups) : {};
  return { schemas, dictionaries, eventHooks, impactRules, signalRules, groups };
}
```

```javascript
function ids(list = []) {
  return new Set((Array.isArray(list) ? list : []).map(item => item?.id).filter(Boolean));
}

function add(issues, code, message, path) {
  issues.push({ severity: 'error', code, message, path });
}

export function validateGameData(configs = {}, options = {}) {
  const issues = [];
  const effectIds = ids(configs.effects?.effects);
  const abilityIds = ids(configs.abilities?.abilities);
  const itemIds = ids(configs.itemDefs?.items);
  const tagIds = new Set(configs.tags?.tags || configs.tags || []);

  for (const effect of configs.effects?.effects || []) {
    if (!String(effect.id || '').startsWith('ge_')) {
      add(issues, 'bad_ge_prefix', `GE id 必须使用 ge_ 前缀：${effect.id}`, `effects.${effect.id}`);
    }
    for (const tag of effect.grantsTags || []) {
      if (!tagIds.has(tag) && !wildcardAllows(tagIds, tag)) {
        add(issues, 'missing_tag', `GE 引用缺失 Tag：${tag}`, `effects.${effect.id}.grantsTags`);
      }
    }
  }

  for (const ability of configs.abilities?.abilities || []) {
    if (!String(ability.id || '').startsWith('ga_')) {
      add(issues, 'bad_ga_prefix', `GA id 必须使用 ga_ 前缀：${ability.id}`, `abilities.${ability.id}`);
    }
    for (const item of ability.requiredItems || []) {
      if (item.itemId && !itemIds.has(item.itemId)) {
        add(issues, 'missing_required_item', `GA requiredItems 引用缺失物品：${item.itemId}`, `abilities.${ability.id}.requiredItems`);
      }
    }
    for (const entry of ability.grantsEffects || []) {
      const effectId = typeof entry === 'string' ? entry : entry.effect;
      if (!effectIds.has(effectId)) {
        add(issues, 'missing_granted_effect', `GA grantsEffects 引用缺失 Effect：${effectId}`, `abilities.${ability.id}.grantsEffects`);
      }
    }
  }

  for (const item of configs.itemDefs?.items || []) {
    for (const spec of item.effects || item.properties?.effects || []) {
      if (!effectIds.has(spec.effect)) {
        add(issues, 'missing_item_effect', `物品 effects 引用缺失 Effect：${spec.effect}`, `items.${item.id}.effects`);
      }
    }
    for (const abilityId of item.grantsAbilities || []) {
      if (!abilityIds.has(abilityId)) {
        add(issues, 'missing_item_ability', `物品 grantsAbilities 引用缺失 Ability：${abilityId}`, `items.${item.id}.grantsAbilities`);
      }
    }
  }

  if (options.strict && issues.length > 0) {
    const text = issues.map(issue => `${issue.path}: ${issue.message}`).join('; ');
    throw new Error(`game data validation failed: ${text}`);
  }
  return issues;
}

function wildcardAllows(tagIds, tag) {
  for (const registered of tagIds) {
    if (String(registered).endsWith('.*')) {
      const prefix = registered.slice(0, -1);
      if (String(tag).startsWith(prefix)) return true;
    }
  }
  return false;
}
```

```javascript
export function evaluateCondition(condition = {}, getValue) {
  const actual = getValue(condition.key);
  switch (condition.op) {
    case 'lt': return actual < condition.value;
    case 'lte': return actual <= condition.value;
    case 'gt': return actual > condition.value;
    case 'gte': return actual >= condition.value;
    case 'eq': return actual === condition.value;
    case 'neq': return actual !== condition.value;
    case 'exists': return actual != null;
    default: return false;
  }
}

export function evaluateConfigRules(config = {}, getValue) {
  let priority = Number(config.basePriority || 0);
  let urgency = 0;
  for (const rule of config.rules || []) {
    if (evaluateCondition(rule.condition, getValue)) {
      priority += Number(rule.priorityBoost || 0);
      urgency += Number(rule.urgencyBoost || 0);
    }
  }
  const satisfied = config.satisfiedCondition
    ? evaluateCondition(config.satisfiedCondition, getValue)
    : false;
  return { priority, urgency, satisfied };
}
```
