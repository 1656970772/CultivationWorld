# 妖兽伤害、斩妖 Job 与数值修为 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将旧 NPC 直接执行型 Action 迁移为 Job/Toil，并实现统一战斗公式、真实斩妖任务、NPC 战斗智能、历练修为与数值修为。

**Architecture:** GOAP 仍通过 Action 规划，但 NPC 新行为只使用 `executionKind:"job"` 的 JobAction。真实执行下沉到 Job/Toil，旧 Action executor 的业务逻辑先抽成领域服务，再由 Toil 调用，随后删除无引用旧 executor。战斗、任务、历练修为和突破检查都通过共享服务统一入口结算。

**Tech Stack:** 原生 JavaScript ESM、JSON 数据配置、现有 GOAP/Job/Toil 框架、Node 工具测试、真实世界模拟工具。

Create `apps/game/tools/test-npc-action-job-migration.mjs`:

```js
#!/usr/bin/env node
import { readFileSync } from 'node:fs';
import { dirname, resolve } from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = dirname(fileURLToPath(import.meta.url));
const GAME_ROOT = resolve(__dirname, '..');
const load = (p) => JSON.parse(readFileSync(resolve(GAME_ROOT, p), 'utf-8'));

let failed = 0;
function assert(cond, msg) {
  if (!cond) { console.error('  FAIL:', msg); failed++; }
  else console.log('  OK:', msg);
}

const migrated = [
  'act_npc_cultivate',
  'act_npc_train_chamber',
  'act_npc_heal',
  'act_npc_explore',
  'act_npc_accept_hunt_quest',
  'act_npc_accept_quest',
  'act_npc_do_quest',
  'act_npc_turn_in_quest',
  'act_npc_redeem_qi_pill',
  'act_npc_use_qi_pill',
  'act_npc_hunt_enemy',
  'act_npc_kill_enemy',
  'act_npc_teach_disciple',
  'act_npc_visit_master'
];

const replacement = new Map([
  ['act_npc_cultivate', 'act_npc_job_cultivate'],
  ['act_npc_train_chamber', 'act_npc_job_train_chamber'],
  ['act_npc_heal', 'act_npc_job_heal'],
  ['act_npc_explore', 'act_npc_job_explore'],
  ['act_npc_accept_hunt_quest', 'act_npc_accept_monster_hunt_job'],
  ['act_npc_accept_quest', 'act_npc_accept_quest_job'],
  ['act_npc_do_quest', 'act_npc_execute_quest_job'],
  ['act_npc_turn_in_quest', 'act_npc_turn_in_quest_job'],
  ['act_npc_redeem_qi_pill', 'act_npc_job_redeem_qi_pill'],
  ['act_npc_use_qi_pill', 'act_npc_job_use_qi_pill'],
  ['act_npc_hunt_enemy', 'act_npc_job_hunt_enemy'],
  ['act_npc_kill_enemy', 'act_npc_job_kill_enemy'],
  ['act_npc_teach_disciple', 'act_npc_job_teach_disciple'],
  ['act_npc_visit_master', 'act_npc_job_visit_master']
]);

const npcActions = load('data/actions/npc-actions.json');
const npcJobActions = load('data/actions/npc-job-actions.json');
const actionSets = load('data/actions/npc-action-sets.json');
const simpleIds = new Set(npcActions.map(a => a.id));
const jobIds = new Set(npcJobActions.map(a => a.id));

for (const oldId of migrated) {
  assert(!simpleIds.has(oldId), `${oldId} removed from npc-actions.json`);
  assert(!actionSets.defaultNpcActionIds.includes(oldId), `${oldId} removed from defaultNpcActionIds`);
  const newId = replacement.get(oldId);
  assert(jobIds.has(newId), `${newId} exists in npc-job-actions.json`);
  assert(actionSets.defaultNpcJobActionIds.includes(newId), `${newId} appears in defaultNpcJobActionIds`);
}

for (const action of npcJobActions) {
  assert(action.executionKind === 'job', `${action.id} is executionKind=job`);
  assert(action.jobId?.startsWith('job_'), `${action.id} references a job_ id`);
}

if (failed > 0) {
  console.error(`\nNPC Action migration guard failed: ${failed}`);
  process.exit(1);
}
console.log('\nNPC Action migration guard passed');
```

export function acceptQuest(entity, worldContext, opts = {}) {
  const questTemplates = worldContext.questTemplates;
  if (!questTemplates) return { success: false, reason: 'quest_templates_missing' };
  const cult = getCultivationConfig(worldContext);
  const rankMaxDifficulty = cult.rankMaxDifficulty || {};
  const rankId = entity.state.get('rankId') || 'mortal';
  const maxDiff = rankMaxDifficulty[rankId] ?? 2;
  const { difficulties, questTypes, randomQuestSpawnChance } = questTemplates;
  const economy = getEconomyConfig(worldContext);
  const available = [];
  for (const qt of questTypes) {
    if (opts.forceMonsterHunt && !isMonsterHuntQuest(qt.id, economy)) continue;
    const [minD, maxD] = qt.difficultyRange;
    const effectiveMax = Math.min(maxD, maxDiff);
    if (minD > effectiveMax) continue;
    for (let d = minD; d <= effectiveMax; d++) {
      const chance = qt.repeatable ? 1 : (randomQuestSpawnChance[String(d)] || 0.5);
      if (worldContext.rng.next() < chance) available.push({ quest: qt, difficulty: d });
    }
  }
  if (available.length === 0) return { success: false, reason: 'no_available_quest' };
  const picked = pickQuestCandidate(entity, worldContext, available, { forceMonsterHunt: !!opts.forceMonsterHunt });
  const diffInfo = difficulties.find(d => d.level === picked.difficulty);
  const questLoc = typeof worldContext.resolveQuestLocation === 'function'
    ? worldContext.resolveQuestLocation(entity, picked.quest, picked.difficulty)
    : null;
  writeQuestState(entity, picked, diffInfo, questLoc);
  return { success: true, picked, diffInfo, questLoc };

export function resolveCombatEncounter(input = {}) {
  const scene = input.scene || 'generic';
  const random = typeof input.random === 'function' ? input.random : Math.random;
  const roll = 0.8 + random() * 0.4;
  const power = Math.max(0, Number(input.power) || 0);
  const defense = Math.max(0, Math.min(0.95, Number(input.defense) || 0));
  const sceneMultiplier = sceneDamageMultiplier(scene, input.worldContext);
  const damage = Math.max(1, Math.round(power * (1 - defense) * roll * sceneMultiplier));
  const winChance = computeWinChance(scene, input);
  const died = false;
  return {
    scene,
    hit: true,
    damage,
    winChance,
    died,
    injuryGain: damage > 0 ? 1 : 0,
    retreatSuggested: shouldSuggestRetreat(scene, input, damage),
    deathInfo: null,
    experienceContext: {
      sourceKind: scene,
      value: Number(input.value) || 0,
      riskScore: Number(input.riskScore) || 0,
      outcome: 'success',
    },
  };
}

function sceneDamageMultiplier(scene, worldContext) {
  if (scene === 'monster_ambush') {
    return worldContext?.balanceConfig?.monsterSpawn?.combat?.damageMultiplier
      ?? worldContext?.monsterSpawn?.combat?.damageMultiplier
      ?? 1;
  }
  const sceneCfg = worldContext?.balanceConfig?.combat?.encounterScenes?.[scene];
  return sceneCfg?.damageMultiplier ?? 1;
}

function computeWinChance(scene, input) {
  if (scene !== 'monster_hunt_quest' && scene !== 'pvp' && scene !== 'quest_risk') return null;
  const attackPower = Math.max(0.01, Number(input.power) || 0.01);
  const defensePower = Math.max(0.01, Number(input.defenderPower ?? input.monsterPower) || 0.01);
  return Math.max(0.02, Math.min(0.98, attackPower / (attackPower + defensePower)));
}

function shouldSuggestRetreat(scene, input, damage) {
  if (scene !== 'monster_ambush' && scene !== 'monster_hunt_quest') return false;
  const maxHp = Number(input.defender?.state?.get?.('maxHp') ?? input.defenderMaxHp ?? 0);
  return maxHp > 0 && damage >= maxHp * 0.4;
}

import { ToilExecutor, ToilResultStatus } from '../../abstract/toil.js';

function write(entity, key, value) {
  if (typeof entity?.state?.set === 'function') entity.state.set(key, value);
}

function read(entity, key, fallback = 0) {
  const value = typeof entity?.state?.get === 'function' ? entity.state.get(key) : entity?.state?.[key];
  return value ?? fallback;
}

export class NPCAssessCombatRiskToilExecutor extends ToilExecutor {
  run(entity, worldContext, job) {
    const monster = job?.context?.monster || (job?.context?.monsterId ? worldContext?.entityRegistry?.getById?.(job.context.monsterId) : null);
    const npcPower = typeof worldContext?.npcCombatPower === 'function' ? worldContext.npcCombatPower(entity) : 1;
    const monsterPower = Number(monster?.state?.get?.('power') ?? monster?.power ?? 1);
    const injury = Number(read(entity, 'injuryLevel', 0));
    const hp = Number(read(entity, 'hp', 1));
    const maxHp = Math.max(1, Number(read(entity, 'maxHp', 1)));
    const riskScore = monsterPower / Math.max(1, npcPower) + injury * 0.2 + (hp / maxHp < 0.35 ? 1 : 0);
    write(entity, 'combatRiskScore', riskScore);
    write(entity, 'monsterTooDangerous', riskScore >= 8);
    write(entity, 'shouldRetreat', hp / maxHp < 0.35 || injury >= 3);
    write(entity, 'needsCombatSupply', entity.inventory?.getAmount?.('pill_rejuvenation') <= 0);
    write(entity, 'combatReady', riskScore < 8 && hp / maxHp >= 0.35);
    if (riskScore >= 8) return { status: ToilResultStatus.REPLAN, reason: 'monster_too_dangerous', contextPatch: { combatRiskScore: riskScore } };
    return { status: ToilResultStatus.SUCCESS, reason: 'combat_risk_assessed', contextPatch: { combatRiskScore: riskScore } };
  }
}

export class NPCPrepareCombatSupplyToilExecutor extends ToilExecutor {
  run(entity) {
    const hasHeal = (entity.inventory?.getAmount?.('pill_rejuvenation') || 0) > 0;
    write(entity, 'needsCombatSupply', !hasHeal);
    write(entity, 'combatReady', hasHeal);
    return hasHeal
      ? { status: ToilResultStatus.SUCCESS, reason: 'combat_supply_ready' }
      : { status: ToilResultStatus.REPLAN, reason: 'combat_supply_missing' };
  }
}

export class NPCRetreatToSafePlaceToilExecutor extends ToilExecutor {
  run(entity, worldContext) {
    const target = worldContext?.resolveTarget?.(entity, 'safe_retreat') || worldContext?.resolveTarget?.(entity, 'faction_hq');
    if (!entity.spatial || !target) {
      write(entity, 'shouldRetreat', false);
      return { status: ToilResultStatus.SUCCESS, reason: 'retreat_no_spatial' };
    }
    if (entity.spatial.tileX === target.x && entity.spatial.tileY === target.y) {
      write(entity, 'shouldRetreat', false);
      return { status: ToilResultStatus.SUCCESS, reason: 'retreat_arrived' };
    }
    entity.spatial.setDestination(target.x, target.y);
    return { status: ToilResultStatus.RUNNING, reason: 'retreating' };
  }
}

export class NPCUseHealItemToilExecutor extends ToilExecutor {
  run(entity) {
    if ((entity.inventory?.getAmount?.('pill_rejuvenation') || 0) > 0) {
      entity.inventory.remove('pill_rejuvenation', 1);
    }
    const injury = Number(read(entity, 'injuryLevel', 0));
    write(entity, 'injuryLevel', Math.max(0, injury - 1));
    return { status: ToilResultStatus.SUCCESS, reason: 'heal_item_used' };
  }
}

export class NPCAbortOverdangerousTargetToilExecutor extends ToilExecutor {
  run(entity) {
    write(entity, 'questTargetMonsterId', null);
    write(entity, 'monsterTooDangerous', true);
    return { status: ToilResultStatus.ABORT, reason: 'overdangerous_target_aborted' };
  }
}

export function applyCultivationExperience(entity, worldContext, input = {}) {
  const cfg = worldContext?.balanceConfig?.cultivation?.experience || {};
  if (cfg.enabled === false || !entity?.state) return { gain: 0, reason: 'experience_disabled' };
  const sourceKind = input.sourceKind || 'quest_progress';
  const base = Number(cfg.baseBySource?.[sourceKind] ?? 0);
  if (base <= 0) return { gain: 0, reason: 'experience_source_unconfigured' };
  const valueScale = Math.max(1, Number(cfg.valueScale) || 500);
  const valueMultiplier = Math.min(Number(cfg.maxValueMultiplier) || 3, 1 + Math.log1p(Math.max(0, Number(input.value) || 0)) / Math.log1p(valueScale));
  const riskMultiplier = Math.min(Number(cfg.maxRiskMultiplier) || 3, 1 + Math.max(0, Number(input.riskScore) || 0) * (Number(cfg.riskWeight) || 0));
  const durationMultiplier = Math.min(Number(cfg.maxDurationMultiplier) || 2.5, 1 + Math.log1p(Math.max(1, Number(input.durationDays) || 1)) / Math.log1p(90));
  const outcomeMultiplier = Number(cfg.outcomeMultiplier?.[input.outcome || 'success'] ?? 1);
  const gain = Number((base * valueMultiplier * riskMultiplier * durationMultiplier * outcomeMultiplier).toFixed(4));
  const currentExperience = Number(entity.state.get('experienceCultivation') || 0);
  const currentCultivation = Number(entity.state.get('cultivation') || 0);
  entity.state.set('experienceCultivation', currentExperience + gain);
  entity.state.set('totalCultivation', currentCultivation + currentExperience + gain);
  return { gain, sourceKind, valueMultiplier, riskMultiplier, durationMultiplier, outcomeMultiplier };
}
```

export function nextCultivationRank(entityOrRankId, ranks = []) {
  const rankId = typeof entityOrRankId === 'string' ? entityOrRankId : entityOrRankId?.state?.get?.('rankId');
  const current = ranks.find(r => r.id === rankId);
  const currentOrder = current?.order ?? 0;
  return ranks
    .filter(r => r.category === 'cultivation' || Number.isFinite(r.cultivationRequired))
    .sort((a, b) => (a.order || 0) - (b.order || 0))
    .find(r => (r.order || 0) > currentOrder) || null;
}

export function nextCultivationRequired(entity, ranks = []) {
  return nextCultivationRank(entity, ranks)?.cultivationRequired ?? 100;
}

export function migrateProgressToNumericCultivation(entity, ranks = []) {
  const required = nextCultivationRequired(entity, ranks);
  const progress = Number(entity.state.get('cultivationProgress') || 0);
  const insight = Number(entity.state.get('insight') || 0);
  if (entity.state.get('cultivation') == null) entity.state.set('cultivation', progress * required);
  if (entity.state.get('experienceCultivation') == null) entity.state.set('experienceCultivation', insight * required);
  syncTotalCultivation(entity, ranks);
}

export function syncTotalCultivation(entity, ranks = []) {
  const cultivation = Number(entity.state.get('cultivation') || 0);
  const experience = Number(entity.state.get('experienceCultivation') || 0);
  const total = cultivation + experience;
  entity.state.set('totalCultivation', total);
  const required = nextCultivationRequired(entity, ranks);
  entity.state.set('cultivationProgressRatio', required > 0 ? total / required : 0);
  entity.state.set('totalProgress', required > 0 ? total / required : 0);
  return total;
}

export function canAttemptBreakthrough(entity, ranks = [], cultivationConfig = {}) {
  const next = nextCultivationRank(entity, ranks);
  if (!next) return false;
  const total = Number(entity.state.get('totalCultivation') || 0);
  const cultivation = Number(entity.state.get('cultivation') || 0);
  const qi = Number(entity.state.get('qi') || 0);
  const minRatio = cultivationConfig.minCultivationRatio ?? 0.3;
  return total >= (next.cultivationRequired || 0)
    && cultivation >= (next.cultivationRequired || 0) * minRatio
    && qi >= (next.qiRequired || 0);
}
```

```markdown
# ADR-051：NPC Action Job 化与数值修为

> 日期：2026-06-05
> 状态：已采纳

## 背景

旧 NPC 直接执行型 Action 与 Job/Toil 并存，导致新增行为可能绕过新系统；修炼进度仍是 0..1 比例，不符合“修为像真气一样是明确数值”的设计。

## 决策

NPC 新行为只通过 JobAction + Job + Toil 执行。旧直接执行型 NPC Action 在迁移后删除配置、默认引用和无引用 executor。修炼进度迁移为 `cultivation`、`experienceCultivation`、`totalCultivation` 三个数值字段，`totalProgress` 仅作兼容派生。

## 后果

行为扩展集中到 Job/Toil；战斗、任务和历练可以共享上下文与验证。数值修为可以和真气并列展示并用于突破门槛。
```
